  <script>
    // Dimensions from sentimentsource.csv
    let dimensions = [];
    // Descriptions from descriptions1.csv: { code -> text }
    let descriptionsMap = {};

    const errorBanner = document.getElementById("errorBanner");
    const interpEl = document.getElementById("interpretationContent");

    window.addEventListener("load", () => {
      loadDimensionsCsv();
      loadDescriptionsCsv();
    });

    async function loadDimensionsCsv() {
      try {
        const resp = await fetch("sentimentsource.csv", { cache: "no-store" });
        if (!resp.ok) throw new Error("HTTP status " + resp.status);
        const text = await resp.text();
        dimensions = parseDimensionsCsv(text);

        if (dimensions.length !== 4) {
          console.warn("sentimentsource.csv rows:", dimensions.length, "expected 4");
        }

        const scores = buildInitialScores();
        drawRadarChart("radarCanvas", scores);
        updateInterpretation(scores);
      } catch (err) {
        console.error("Error loading sentimentsource.csv:", err);
        dimensions = [];
        const scores = buildInitialScores();
        drawRadarChart("radarCanvas", scores);
        updateInterpretation(scores);
      }
    }

    async function loadDescriptionsCsv() {
      try {
        const resp = await fetch("descriptions1.csv", { cache: "no-store" });
        if (!resp.ok) throw new Error("HTTP status " + resp.status);
        const text = await resp.text();
        descriptionsMap = parseDescriptionsCsv(text);
      } catch (err) {
        console.error("Error loading descriptions1.csv:", err);
        descriptionsMap = {};
      }
    }

    function parseDimensionsCsv(csvText) {
      const lines = csvText.split(/\r?\n/).filter(line => line.trim().length > 0);
      const result = [];

      for (const line of lines) {
        const cols = line.split(",").map(c => c.trim());
        if (cols.length < 2) continue;

        const label = cols[0];
        const rawWords = cols.slice(1).filter(w => w.length > 0);

        const words = rawWords.map(w => ({
          raw: w,
          norm: w.toLowerCase().replace(/_/g, " ").replace(/\s+/g, " ").trim()
        }));

        result.push({ label, words });
      }

      return result;
    }

    function parseDescriptionsCsv(csvText) {
      const lines = csvText.split(/\r?\n/).filter(line => line.trim().length > 0);
      const map = {};

      // Detect header
      let startIndex = 0;
      if (/code/i.test(lines[0]) && /text/i.test(lines[0])) {
        startIndex = 1;
      }

      for (let i = startIndex; i < lines.length; i++) {
        const line = lines[i];
        const cols = line.split(",");
        if (cols.length < 2) continue;
        const code = cols[0].trim();
        const text = cols.slice(1).join(",").trim();
        if (code) {
          map[code] = text;
        }
      }

      return map;
    }

    function buildInitialScores() {
      return dimensions.map(d => ({
        label: d.label,
        count: 0,
        totalPossible: d.words.length
      }));
    }

    function normalizeTextForMatching(text) {
      return " " +
        text
          .toLowerCase()
          .replace(/_/g, " ")
          .replace(/[^a-z0-9\s]+/g, " ")
          .replace(/\s+/g, " ")
        + " ";
    }

    function computeScores(userText) {
      const normalizedText = normalizeTextForMatching(userText);
      const results = [];

      for (const dim of dimensions) {
        const uniqueNormWords = new Set(
          dim.words.map(w => w.norm).filter(n => n.length > 0)
        );
        const totalPossible = uniqueNormWords.size || 1;

        let count = 0;
        for (const normWord of uniqueNormWords) {
          const pattern = " " + normWord + " ";
          if (normalizedText.includes(pattern)) {
            count += 1;
          }
        }

        results.push({
          label: dim.label,
          count,
          totalPossible
        });
      }

      return results;
    }

    function drawRadarChart(canvasId, scores) {
      const canvas = document.getElementById(canvasId);
      if (!canvas || !canvas.getContext) return;
      const ctx = canvas.getContext("2d");

      ctx.clearRect(0, 0, canvas.width, canvas.height);

      const width = canvas.width;
      const height = canvas.height;
      const centerX = width / 2;
      const centerY = height / 2;
      const radius = Math.min(width, height) * 0.38;

      const numAxes = scores.length || 4;
      const angleStep = (2 * Math.PI) / numAxes;

      // Raw max match count
      let rawMax = 0;
      scores.forEach(s => {
        if (s.count > rawMax) rawMax = s.count;
      });

      // Color band from raw max
      let baseColor, fillColor, gridColor, axisColor, pointColor;
      if (rawMax < 5) {
        baseColor  = "#ef4444";
        fillColor  = "rgba(239,68,68,0.25)";
        gridColor  = "rgba(127,29,29,0.9)";
        axisColor  = "rgba(127,29,29,0.9)";
        pointColor = "#fecaca";
      } else if (rawMax < 20) {
        baseColor  = "#facc15";
        fillColor  = "rgba(250,204,21,0.25)";
        gridColor  = "rgba(133,77,14,0.85)";
        axisColor  = "rgba(133,77,14,0.85)";
        pointColor = "#fef3c7";
      } else {
        baseColor  = "#22c55e";
        fillColor  = "rgba(34,197,94,0.25)";
        gridColor  = "rgba(22,101,52,0.9)";
        axisColor  = "rgba(22,101,52,0.9)";
        pointColor = "#bbf7d0";
      }

      // Avoid divide-by-zero
      const effectiveMax = rawMax > 0 ? rawMax : 1;

      // Background
      const radialBg = ctx.createRadialGradient(
        centerX, centerY, 0,
        centerX, centerY, radius * 1.25
      );
      radialBg.addColorStop(0, "rgba(15,23,42,0.4)");
      radialBg.addColorStop(1, "rgba(15,23,42,0.95)");
      ctx.fillStyle = radialBg;
      ctx.fillRect(0, 0, width, height);

      // Grid rings
      ctx.strokeStyle = gridColor;
      ctx.lineWidth = 1;
      const rings = 4;
      for (let i = 1; i <= rings; i++) {
        const levelValue = (effectiveMax / rings) * i;
        const r = radius * (levelValue / effectiveMax);
        ctx.beginPath();
        ctx.arc(centerX, centerY, r, 0, 2 * Math.PI);
        ctx.stroke();
      }

      // Axes + labels
      ctx.font = "11px system-ui, -apple-system, BlinkMacSystemFont, 'Segoe UI', sans-serif";
      ctx.fillStyle = "#cbd5f5";

      scores.forEach((s, i) => {
        const angle = -Math.PI / 2 + i * angleStep;
        const x = centerX + radius * Math.cos(angle);
        const y = centerY + radius * Math.sin(angle);

        // Axis line
        ctx.strokeStyle = axisColor;
        ctx.beginPath();
        ctx.moveTo(centerX, centerY);
        ctx.lineTo(x, y);
        ctx.stroke();

        // Label
        const labelRadius = radius * 1.12;
        const lx = centerX + labelRadius * Math.cos(angle);
        const ly = centerY + labelRadius * Math.sin(angle);

        ctx.textAlign = Math.abs(Math.cos(angle)) < 0.1
          ? "center"
          : (Math.cos(angle) > 0 ? "left" : "right");
        ctx.textBaseline = Math.abs(Math.sin(angle)) < 0.1
          ? "middle"
          : (Math.sin(angle) > 0 ? "top" : "bottom");

        ctx.fillText(s.label, lx, ly);
      });

      // Data polygon
      ctx.beginPath();
      scores.forEach((s, i) => {
        const value = Math.max(0, s.count || 0);
        const angle = -Math.PI / 2 + i * angleStep;
        const r = radius * (value / effectiveMax);
        const x = centerX + r * Math.cos(angle);
        const y = centerY + r * Math.sin(angle);
        if (i === 0) ctx.moveTo(x, y); else ctx.lineTo(x, y);
      });
      ctx.closePath();

      ctx.strokeStyle = baseColor;
      ctx.lineWidth = 2.2;
      ctx.stroke();

      ctx.fillStyle = fillColor;
      ctx.fill();

      // Points
      ctx.fillStyle = pointColor;
      scores.forEach((s, i) => {
        const value = Math.max(0, s.count || 0);
        const angle = -Math.PI / 2 + i * angleStep;
        const r = radius * (value / effectiveMax);
        const x = centerX + r * Math.cos(angle);
        const y = centerY + r * Math.sin(angle);
        ctx.beginPath();
        ctx.arc(x, y, 4, 0, 2 * Math.PI);
        ctx.fill();
      });
    }

    function updateInterpretation(scores) {
      if (!scores || scores.length === 0) {
        interpEl.textContent = "Your approach to AI in the workforce is even.";
        return;
      }

      const counts = scores.map(s => s.count);
      const minCount = Math.min(...counts);

      if (minCount === 0 && counts.every(c => c === 0)) {
        interpEl.textContent = "Your approach to AI in the workforce is even.";
        return;
      }

      // Build pattern of H/L
      const patternArr = counts.map(c => (c >= 2 * minCount ? "H" : "L"));
      const pattern = patternArr.join("");

      if (!pattern.includes("H")) {
        interpEl.textContent = "Your approach to AI in the workforce is even.";
        return;
      }

      const desc = descriptionsMap[pattern];
      if (desc) {
        interpEl.innerHTML = `<span class="interpretation-code">${pattern}</span> – ${desc}`;
      } else {
        interpEl.innerHTML = `<span class="interpretation-code">${pattern}</span> – No specific description found for this pattern.`;
      }
    }

    function analyzeText() {
      errorBanner.style.display = "none";
      if (!dimensions || dimensions.length === 0) {
        errorBanner.textContent = "Unable to analyze right now. Please check that the page is correctly loaded.";
        errorBanner.style.display = "block";
        return;
      }
      const text = document.getElementById("textInput").value.trim();
      const scores = computeScores(text);
      drawRadarChart("radarCanvas", scores);
      updateInterpretation(scores);
    }

    function clearText() {
      document.getElementById("textInput").value = "";
      errorBanner.style.display = "none";
      const scores = buildInitialScores();
      drawRadarChart("radarCanvas", scores);
      updateInterpretation(scores);
    }
  </script>
